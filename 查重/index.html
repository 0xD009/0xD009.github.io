# 实验内容概述

利用所学的字符串性质及其相关函数，求两篇英文文章的重复率

# 实现思路

读出两篇文章，存入字符串类型中，对两篇文章分句，逐句比较，然后通过一个与句子权重有关的权值求加权平均数，得到两篇文章的相似度

# 需要考虑的问题

  1. 文件读取，怎样防止溢出或者占用过多不必要的空间
  2. 分句以什么标准，如何操作
  3. 怎样进行两个句子的比较
  4. 怎样定义一个合理的 “相似度”

# 解决方法

  1. 针对文件读取问题，笔者利用 fseek , ftell , rewind 三个函数，在读取之前获得文章的长度，并重置文件指针
  2. 分句给出两段处理，首先我们有 punc 数组，存放用于划分意群的字符，其他不足以划分意群的字符会存放在 blank 数组中，在对句子预处理（见 preswitch() 函数）时将它们删除，避免影响相似度
  3. 两个句子的比较，可以采用矩阵法，矩阵中的元素 aija_{ij}aij​ 代表 A 句子的第 iii 个字符是否与 B 句子的第 jjj 个字符相同，然后求得这个矩阵最长的对角线（见 longest_diag() 函数），作为相似字符数 nnn
  4. 给出如下的相似度定义：
\begin{align*} & similarity = \frac{\sum\limits_{i \geq 1} \max\limits_{j \geq 1}\{n_{ij}\}}{weight} \\ \\ & 其中n_{ij}指与文章A中i句子与文章B中j句子的相同字符数\\ \\ & weight指文章M去掉所有标点剩余有效内容长度 \end{align*}

# 其他设计

  1. 对于句子的划分，采用直接把标点置为 \0 （见 split2sentences() 函数）的方法，这将影响程序以句子为单位遍历文章，为此笔者准备了 shift() 函数专门用于处理指针的移动
  2. 对于相似字符小于等于 5 的情况，在常识下，我们不认为那是相似的，所以在 longest_diag() 函数中我们的返回值为 (res > 5) ? res : 0
  3. 在源码的各处设置了一些打印函数，这样有助于 debug ，也方便看到程序的执行效果

# 源码

#include<stdio.h>
#include<stdlib.h>
#include<string.h>

#define MAXSIZE 100000
#define SPLITNUM 5
#define BLANKNUM 6

char punc[] = {
	'.', '?', '!', ';', '\n'
};
char blank[] = {
	'\'', ':', ',', ' ', '\"', '-'
};
int shift(char *, int, int); //移动指针
char *read_message(const char *, int &); //读取整篇文章
void split2sentences(char *); //把纹章拆成句子
void preswitch(char *, int &); //预处理句子
int longest_diag(int **, int, int); //求矩阵的最长对角线
int cmp_sentences(char *, char *); //比较两个句子中最长的共同子串
void cmp_messages(const char *, const char *); //比较两篇文章

int main() {
	//cmp_messages("hamlet.txt", "JaneEyre.txt");
	cmp_messages("test1.txt", "test2.txt");

	return 0;
}

int shift(char *message, int p, int len) { //传入len，避免溢出
	int l = strlen(message + p);
	while(*(message + p + l) == 0 && p + l < len) l++;
	return p + l;
}

char *read_message(const char *filename, int &len) {
	FILE *file = fopen(filename, "r");
	if (file == NULL) { //考虑鲁棒性
		printf("Invalid file!");
		exit(114514);
	}
	
	fseek(file, 0, SEEK_END);
    len = ftell(file); //获取文章长度
	if (len > MAXSIZE) { //同考虑鲁棒性
		printf("Overlong!");
		exit(1919810);
	}
    char *message = (char*)malloc((len + 1) * sizeof(char)); 
    rewind(file);  //重置指针
    fread(message, sizeof(char), len, file);
	message[len] = '\0';

	fclose(file); //别忘了

	return message;
}

void split2sentences(char *message) {
	for (; *message != 0; message++) {
		for (int i = 0; i < SPLITNUM; i ++ ) {
			if (*message == punc[i]) *message = 0;
		}
	}
}

void preswitch(char *sentence, int &weight) {
	int len = strlen(sentence);
	weight = len;
	for (int i = 0; i < len; i ++ ) {
		for (int j = 0; j < BLANKNUM; j ++ ) {
			if (sentence[i] == blank[j]) {
				for (int k = i; k < len; k ++ ) //采用经典移位删除，同时不影响句子所占的空间大小
					sentence[k] = sentence[k + 1];
				sentence[len] = '\0';
				weight--;
				break;
			}
		}
	}
}

int longest_diag(int **martix, int a, int b) {
	int buf, res = 0;
	for (int i = 0; i < a; i ++ ) { //遍历从第一行开始的对角线
		buf = 0;
		for (int j = 0; i + j < b && j < a; j ++ ) {
			if (martix[i+j][j] == 1) buf++;
		}
		if (buf > res) res = buf;
	}
	for (int i = 0; i < b; i ++ ) { //遍历从第一列开始的对角线
		buf = 0;
		for (int j = 0; i + j < a && j < b; j ++ ) {
			if (martix[j][i+j] == 1) buf++;
		}
		if (buf > res) res = buf;
	}

	return (res > 5) ? res : 0;
}

int cmp_sentences(char *sentenceA, char *sentenceB) {
	int lenSA = strlen(sentenceA);
	int lenSB = strlen(sentenceB);

	//生成矩阵
	int **martix = (int **)malloc(sizeof(int *) * lenSB);
	for (int i = 0; i < lenSB; i ++ ) 
		*(martix + i) = (int *)malloc(sizeof(int) * lenSA);
	for (int i = 0; i < lenSB; i ++ ) {
		for (int j = 0; j < lenSA; j ++ ) {
			if (*(sentenceA + j) == *(sentenceB + i))
				martix[i][j] = 1;
			else martix[i][j] = 0;
		}		
	}

	//打印矩阵
	printf("Martix:\n");
	printf("  ");
	for (int i = 0; i < lenSA; i ++ ) printf("%2c", *(sentenceA + i));
	printf("\n");
	for (int i = 0; i < lenSB; i ++ ) {
		printf("%2c", *(sentenceB + i));
		for (int j = 0; j < lenSA; j ++ ) { 
			printf("%2d", martix[i][j]);
		}
		printf("\n");
	}

	printf("longest:%d\n", longest_diag(martix, lenSA, lenSB));
	int res = longest_diag(martix, lenSA, lenSB);

	//养成好习惯
	for (int i = 0; i < lenSB; i ++ ) free(*(martix + i));
	free(martix);

	return res;
}


void cmp_messages(const char* filenameA, const char* filenameB) {
	int lenA, lenB;

	char* messageA = read_message(filenameA, lenA);
	char* messageB = read_message(filenameB, lenB);

	printf("%s\n", messageA); 
	printf("%s\n", messageB); 

	printf("lenA and lenB: %d %d\n\n", lenA, lenB);

	split2sentences(messageA);
	split2sentences(messageB);
	
	int p2A = 0, p2B = 0, weightA = 0, weightB = 0, weight = 0, sum = 0, max = 0, buf = 0;

	while (p2A < lenA) {
		preswitch(messageA + p2A, weightA);
		while (p2B < lenB) {
			printf("ptr position: %d, %d\n", p2A, p2B);
			
			preswitch(messageB + p2B, weightB);

			printf("from A:"); puts(messageA + p2A);
			printf("from B:"); puts(messageB + p2B);

			buf = cmp_sentences(messageA + p2A, messageB + p2B);
			if (buf > max) max = buf; //取最大的n

			p2B = shift(messageB, p2B, lenB);
		}

		sum += max;
		weight += weightA;
		weightA = 0; max = 0;
		p2A = shift(messageA, p2A, lenA);
		p2B = 0;
	}
	
	printf("Similarity: %.2f", 1.0 * sum / weight);
}
