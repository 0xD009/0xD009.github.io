# 实验内容概述

利用所学的栈及其常用操作求解二维平面内的迷宫问题

# 实现思路

穷举，对路径进行遍历，直到找到出口

# 需要考虑的问题

  1. 避免走重复的路径
  2. 避免走出迷宫地界
  3. 遇到死路后如何实现回退
  4. 迷宫无解如何处理

# 解决办法

  1. 对路径上每一点构造方向栈，走过该方向则出栈，避免方向选择重复从而避免路径重复
  2. 进行走法合法性检验时，在原迷宫数组外围加上一圈 “栅栏”
  3. 构造路径栈，每走一步则入栈，遇到死路或方向栈为空后则出栈
  4. 检查路径栈，若为空，则说明迷宫无解

# 其他设计

  1. 实验源码在 linux 系统下编译
  2. 构造位置结构体，存有玩家目前的坐标，同时由于方向栈结构简单，为图方便，也一同放入此结构体中
  3. 程序在终端中动态执行，可以直观地看到探索迷宫的过程
  4. 重要函数 Move() ， Look() 等的相关说明见下面的源码

//位置结构体
typedef struct Position{
	int x;
	int y;
	int Dstack[4][2];
	int StackSize;
}Position;

//路径栈
typedef struct Path{
	int size;
	Position *stack[256];
}Path;

//路径栈出栈函数
Position *Pathpop(Path *path) {
    Position *pos = path -> stack[path -> size - 1];
    path -> size -= 1;
    return pos;
}

//路径栈入栈函数
void Pathpush(Path *path, Position *curpos) {
    path -> size += 1;
    path -> stack[path -> size - 1] = curpos;
}

//位置结构体初始化函数
void InitPosition(Position *pos) {
    int d[4][2] = {{0,1},{0,-1},{1,0},{-1,0}};
    for(int i = 0; i < 4; i++) {
        for (int j = 0; j < 2; j++) pos -> Dstack[i][j] = d[i][j];
    }//方向栈内是四个表示方向的二元组
    pos -> StackSize = 4;
}

//获取方向，内含方向栈出栈操作
int *GetDirection(Position *curpos) {
	int *direction = (int *) malloc(sizeof(int) * 2);
	direction = curpos -> Dstack[curpos -> StackSize - 1];
	curpos -> StackSize -= 1;
	return direction;
}

//检测走法是否合理，合理则返回1，否则返回0
int Look(Position *curpos, int *direction) {
    int protected_maze[18][18];
    for(int i = 0; i < 18; i ++) {
        for(int j = 0; j < 18; j ++) {
            if(i ==0 || j == 0) protected_maze[i][j] = '0';
            else if(i == 17 || j == 17) protected_maze[i][j] = '0';
            else protected_maze[i + 1][j + 1] = maze[i][j];
        }
    }//围上栅栏
    char dest = protected_maze[curpos->x + direction[0] + 1][curpos->y + direction[1] + 1];
    if(dest != ' ') return 0; //不为' '包括了墙和来路两种情况
    return 1;
}

//移动函数
Position *Move(Position *curpos, Path *path) {
    int *direction;
    if(curpos -> StackSize) {
	    //不断获取方向，直到方向栈空或走法合理
        do{
            direction = GetDirection(curpos);
        }
        while(!Look(curpos, direction) && curpos -> StackSize);
        //真正开始移动，包含路径栈入栈，迷宫数组变换
        if(Look(curpos, direction)) {
            Position *pos = (Position *)malloc(sizeof(Position));
            pos -> x = curpos -> x + direction[0];
            pos -> y = curpos -> y + direction[1];
            InitPosition(pos);
            Pathpush(path, curpos);
            curpos = pos;
            maze[curpos -> x][curpos -> y] = '+';
            return curpos;
        }
    }
	//无路可走，包含路径栈出栈，迷宫数组变换
    maze[curpos -> x][curpos -> y] = ' ';
    curpos = Pathpop(path);
    return curpos;
}

//实现屏幕的更新
void UpdataScreen() {
    system("clear");
    for(int i = 0; i < 16; i ++) {
		for(int j = 0; j < 16; j ++) printf("%3c", maze[i][j]);
		printf("\n");
	}
    printf("\n");
    usleep(300000); //单位为微秒，提升观感
}

int main() {
	//初始化和起点设置
    Position *curpos = (Position *)malloc(sizeof(Position));
    Path *path = (Path *)malloc(sizeof(Path));
    InitPosition(curpos);
    curpos -> x = 4;
    curpos -> y = 0;
    path -> size = 0;
    maze[curpos -> x][curpos -> y] = '+';

	//遍历路径直到解出或发现无解
    do {
        curpos = Move(curpos, path);
        UpdataScreen();
    }
    while(!(curpos -> x == 12 && curpos -> y == 15 || path -> size == 0));

    if (path -> size == 0) printf("STUCK!"); //无解
    else printf("OUT!"); //解出

    return 0;
}

# 心得体会
