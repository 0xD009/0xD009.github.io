# 实验内容概述

对全英小写的文本文件进行哈夫曼编码，并将每个字符的编码值写入文件

# 实现思路

基本已经被老师讲的七七八八了，而且书上有很全的代码，我们只需要实现一些细枝末节

# 需要考虑的问题

  1. 文件读写
  2. 怎样每次找到权值最小的两个结点

# 解决方法

  1. 见 read_message() 、 write_in() 函数
  2. 见 select() 函数

# 其他设计

我们通过 match_weight_and_char() 函数来得到每个字符的权值

# 源码

#include<stdio.h>
#include<stdlib.h>
#include<string.h>

#define MAXSIZE 1000
#define N 26

typedef int WeightType;

typedef struct Stack{
    char elem[MAXSIZE];
    int length;
}Stack;

typedef struct HuffTree{
    int leftchild, rightchild, parent;
    WeightType weight;
}HuffTree;

const char letters[] = "#abcdefghijklmnopqrstuvwxyz";

void stack_init(Stack* stack); //栈初始化
void push(Stack* stack, int value); //压栈
void pop(Stack *stack); //弹栈
char *read_message(const char *filename); //读取整个文件
void match_weight_with_char(char *text, WeightType *weight); //得到字符的权重
void select(HuffTree *tree, int n, int i, int &s1, int &s2); //选出权重最小的两个结点
void hufftree_init(HuffTree *tree, WeightType *weight, int n); //哈夫曼树初始化
void coding(HuffTree *tree, char **code, int root, Stack *stack); //转码
void huffman_coding(HuffTree *tree, char **code, int n); //转码的壳
void write_in(const char * filename, char **code); //写进文件里

int main() {
    char *text = read_message("t1.txt");
    WeightType weight[N + 1] = {0};
    match_weight_with_char(text, weight);

    HuffTree *tree = (HuffTree *)malloc(sizeof(HuffTree) * 2 * N);
    hufftree_init(tree, weight, N);

    char **code = (char **)malloc(sizeof(char *) * (N + 1));
    huffman_coding(tree, code, N);

    write_in("t2.txt", code);

    // huffman_encode("test1.txt", "test3.txt", list, code);
    // huffman_decode("test3.txt", "test4.txt", tree, n, list);

    for (int i = 1; i <= N; i ++ ) printf("%c %d %s\n", letters[i], weight[i], code[i]);

    return 0;
}

void stack_init(Stack *stack) {
    stack -> length = 0;
}

void push(Stack *stack, char c) {
    if (stack -> length + 1 < MAXSIZE) stack -> elem[stack -> length++] = c;
    else {
        printf("Stack Overflow!\n");
        exit(11111);
    }
}

void pop(Stack *stack) {
    if (stack -> length) stack -> length--;
    else {
        printf("Stack Overflow!\n");
        exit(22222);
    }
}

char *read_message(const char *filename) {
	FILE *file = fopen(filename, "r");
	if (file == NULL) { //考虑鲁棒性
		printf("Invalid file!\n");
		exit(33333);
	}
	
	fseek(file, 0, SEEK_END);
    int len = ftell(file); //获取文章长度
	if (len > MAXSIZE) { //同考虑鲁棒性
		printf("Overlong!\n");
		exit(44444);
	}
    char *message = (char*)malloc((len + 1) * sizeof(char)); 
    rewind(file);  //重置指针
    fread(message, sizeof(char), len, file);
	message[len] = '\0';

	fclose(file); //别忘了

	return message;
}

void match_weight_with_char(char *text, WeightType *weight) {
    int w = 0;

    for (int i = 1; i <= N; i ++ ) {
        for (int j = 0; j < strlen(text); j ++ ) {
            if (text[j] == letters[i]) w ++;
        }
        weight[i] = w;
        w = 0;
    }
}

void select(HuffTree *tree, int n, int i, int &s1, int &s2) {
    int w1 = MAXSIZE, w2 = MAXSIZE;
    for (int j = 1; j < i; j ++ ) {
        if (tree[j].weight < w1 && !tree[j].parent) {
            w2 = w1; s2 = s1;
            w1 = tree[j].weight; s1 = j;
        } 
        else if (tree[j].weight < w2 && !tree[j].parent) {
            w2 = tree[j].weight; s2 = j;
        }
    }
}

void hufftree_init(HuffTree *tree, WeightType *weight, int n) {
    int s1, s2, m = 2 * n - 1;

    for (int i = 1; i <= m; i ++ ) {
        tree[i].weight = (i <= n) ? weight[i] : 0;
        tree[i].leftchild = tree[i].rightchild = tree[i].parent = 0;
    }

    for (int i = n + 1; i <= m; i ++ ) {
        select(tree, n, i, s1, s2);
        tree[i].leftchild = s1;
        tree[i].rightchild = s2;
        tree[i].weight = tree[s1].weight + tree[s2].weight;
        tree[s1].parent = tree[s2].parent = i;
    }
}

void coding(HuffTree *tree, char **code, int root, Stack *stack) {
    if (root) {
        if (tree[root].leftchild == 0) {
            push(stack, '\0');
            code[root]  = (char *)malloc(sizeof(char) * stack -> length);
            strcpy(code[root], stack -> elem);
            pop(stack);
        }
        push(stack, '0');
        coding(tree, code, tree[root].leftchild, stack);
        pop(stack);
        push(stack, '1');
        coding(tree, code, tree[root].rightchild, stack);
        pop(stack);
    }
}

void huffman_coding(HuffTree *tree, char **code, int n) {
    Stack S;
    stack_init(&S);

    coding(tree, code, 2 * n - 1, &S);
}

void write_in(const char *filename, char **code) {
    FILE *out = fopen(filename, "w");

    for (int i = 1; i <= N; i++) {
        fwrite(letters + i, sizeof(char), 1, out);
        fwrite("   ", sizeof(char), 3, out);
        fwrite(code[i], sizeof(char), strlen(code[i]), out); 
        fwrite("\n", sizeof(char), 1, out);
    }

    fclose(out);
}
