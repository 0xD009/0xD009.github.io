# Reverse 四题 - 熟悉 IDA 操作

# 1. 攻防世界 easyRE1

IDA+F5 得到如下伪代码：

int __cdecl main(int argc, const char **argv, const char **envp)
{
  char s1[264]; // [rsp+10h] [rbp-110h] BYREF
  unsigned __int64 v5; // [rsp+118h] [rbp-8h]

  v5 = __readfsqword(0x28u);
  puts("What is the password?");
  gets(s1);
  if ( !strcmp(s1, "the password") )
    puts("FLAG:db2f62a36a018bce28e46d976e3f9864");
  else
    puts("Wrong!!");
  return 0;
}

直接读出 flag, 注意 flag 的标准形式: flag{xxxxxxxxxx} , 经修改后提交，解毕.

  • check()函数 在定义处传入参数类型为 _byte * , 而 main()函数 中实际传入的后两个参数为 _dword * .

  • IDA 伪代码中数据类型与 C 语言中的对应
    IDA 伪代码C 语言占用
    _BYTE 或 unsigned __int8unsigned char1 字节
    _WORD 或 unsigned __int16short2 字节
    _DWORDunsigned int4 字节
    _QWORDunsigned long long8 字节
  • 用到的赋值
    char s[4];
*(unsigned int*) s = 0x333231;
    达到的效果是 s[0] = '1' (0x31), s[1] = '2' (0x32), s[2] = '3' (0x33), s[3] = '\x00'

# 4. 信安实践 RC4

IDA+F5 得到如下伪代码:

int __cdecl main(int argc, const char **argv, const char **envp)
{

  _BYTE v5[256]; // [esp+1Eh] [ebp-132h] BYREF
  char Str[50]; // [esp+11Eh] [ebp-32h] BYREF

  __main();
  printf("Input your flag: ");
  scanf("%50s", Str);
  RC4Init((int)v5, (int)&unk_41105B, 0x10u);
  v3 = strlen(Str);
  RC4Crypt(v5, Str, v3);
  if ( !memcmp(Str, &enc_buf, 0x2Cu) )
    puts("Correct!");
  else
    puts("Wrong, try again.");
  return 0;

题设提供加密算法的 C 语言实现如下:

void RC4Init(unsigned char* RC4Box, unsigned char* key, unsigned int keylen) {
	for (int i = 0; i < 256; i++) {
		RC4Box[i] = i;
	}
	for (int i = 0, j = 0; i < 256; i++) {
		j = (j + RC4Box[i] + key[i % keylen]) & 0xff;
		int tmp = RC4Box[i];
		RC4Box[i] = RC4Box[j];
		RC4Box[j] = tmp;
	}
}

void RC4Crypt(unsigned char* RC4Box, char* data, unsigned int len) {
	for (int i = 0, j = 0, k = 0; k < len; k++) {
		i = (i + 1) & 0xff;
		j = (j + RC4Box[i]) & 0xff;
		int tmp = RC4Box[i];
		RC4Box[i] = RC4Box[j];
		RC4Box[j] = tmp;
		data[k] ^= RC4Box[(RC4Box[i] + RC4Box[j]) & 0xff];
	}
}

观察发现:

  • RC4Init()函数 与输入值无关.
  • RC4Crypt()函数 是流加密.
    流加密是对称加密算法的一种，加密和解密双方使用相同伪随机加密数据流（pseudo-randomstream）作为密钥，明文数据每次与密钥数据流顺次对应加密，得到密文数据流。实践中数据通常是一个位（bit）并用异或（xor）操作加密.
