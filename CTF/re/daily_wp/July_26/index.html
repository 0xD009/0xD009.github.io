# Reverse 四题 - 熟悉 IDA 操作

# 1. 信安实践 Have a Cup of TEA

IDA+F5 得到如下伪代码:

unsigned int __cdecl tea_encrypt(unsigned int *a1, _DWORD *a2)
{
  unsigned int result; // eax
  int i; // [esp+10h] [ebp-10h]
  int v4; // [esp+14h] [ebp-Ch]
  unsigned int v5; // [esp+18h] [ebp-8h]
  unsigned int v6; // [esp+1Ch] [ebp-4h]

  v6 = *a1;
  v5 = a1[1];
  v4 = -828624794;
  for ( i = 0; i <= 31; ++i )
  {
    v4 -= 365308511;
    v6 += (v5 + v4) ^ (16 * v5 + *a2) ^ ((v5 >> 5) + a2[1]);
    v5 += (v6 + v4) ^ (16 * v6 + a2[2]) ^ ((v6 >> 5) + a2[3]);
  }
  *a1 = v6;
  result = v5;
  a1[1] = v5;
  return result;
}

int __cdecl main(int argc, const char **argv, const char **envp)
{
  char Buf1[50]; // [esp+1Ah] [ebp-36h] BYREF
  int i; // [esp+4Ch] [ebp-4h]

  __main();
  printf("Input your flag: ");
  memset(Buf1, 0, sizeof(Buf1));
  scanf("%50s", Buf1);
  for ( i = 0; i <= 5; ++i )
    tea_encrypt((unsigned int *)&Buf1[8 * i], key);
  if ( !memcmp(Buf1, enc_buf, 0x30u) )
    puts("Correct!");
  else
    puts("Wrong, try again.");
  return 0;
}

可以看出 flag 经过 tea_encrypt()函数 (每次只加密 8 个字节) 处理后与 enc_buf 在前 48 位上一致，通过 IDA 读出 unsigned char enc_buf[48] 以及 unsigned int key[8] , 在 C 程序中写出 tea_encrypt()函数 的逆:

        v4 += 365308511;

最后求出 flag 为 flag{0h_th1s_cUp_0F_t34_1S_p01sOn!!_c5f13565} , 提交，解毕.

# 2. 信安实践 Treasure Hunter

IDA+F5 得到如下伪代码:

int game()
{
  int v0; // eax
  int v1; // eax
  int v2; // eax
  int v3; // eax
  int result; // eax
  int v5; // [esp+18h] [ebp-10h]
  int v6; // [esp+1Ch] [ebp-Ch]

  v6 = 4;
  v5 = 9;
  if ( maze[94] != 64 || byte_403083 != 84 )
  {
    puts("Too bad, something unexpected.");
    puts("Please report this to your TA.");
    exit(-1);
  }
LABEL_22:
  while ( maze[10 * v5 + v6] != 84 )
  {
    switch ( (unsigned __int8)getchar() )
    {
      case 'a':
        if ( !v6 )
          lost();
        --v6;
        v0 = step++;
        path[v0] = 97;
        goto LABEL_18;
      case 'd':
        if ( v6 == 9 )
          lost();
        ++v6;
        v1 = step++;
        path[v1] = 100;
        goto LABEL_18;
      case 'q':
        puts("Quit.");
        exit(0);
      case 's':
        if ( v5 == 9 )
          lost();
        ++v5;
        v3 = step++;
        path[v3] = 115;
        goto LABEL_18;
      case 'w':
        if ( !v5 )
          lost();
        --v5;
        v2 = step++;
        path[v2] = 119;
LABEL_18:
        if ( maze[10 * v5 + v6] == 35 )
        {
          puts("You tried to hit the wall with your head.");
          die();
        }
        if ( step > 49 )
        {
          puts("You run out of food.");
          die();
        }
        return result;
      default:
        goto LABEL_22;
    }
  }
  return 1;
}

int try_open()
{
  char v1[264]; // [esp+10h] [ebp-108h] BYREF

  RC4Init(v1, path, step);
  RC4Crypt(v1, treasure, 42);
  if ( strncmp(treasure, "flag", 4u) )
    return puts("Treasure has been stolen! Be faster next time!");
  puts("You found a flag!");
  return puts(treasure);
}

int __cdecl main(int argc, const char **argv, const char **envp)
{
  __main();
  puts("Tired of reversing? Let's have a game and try to find the path to the treasure chest!");
  if ( game() )
  {
    puts("Congrats! You found the treasure chest!");
    puts("Opening...");
    try_open();
  }
  return 0;
}

不难发现这真的是一个迷宫游戏，首先要获取解法.
通过观察 game()函数 ，发现游戏操作通过 W , S , A , D 四个键进行操作，迷宫地形由数组 maze[] 决定， v6 和 v5 被赋初值，说明它们分别是两个坐标，初值是起点，由 switch语句 中对二者的限制可知此迷宫为 10X10 规模，从 IDA 中导出数组 maze[] , 发现只有 99 个 char , 将紧接着的 byte_403083 也读出作为第 100 个.
写 C 程序打印迷宫地图方便解迷宫，得到输出入下:

####   ## 
#### #    
     # ###
 ### #    
 ###  ### 
 #### ##
#   #  # #
# # ## #
  #    ##
 ###@####T

其中 #(35) 为迷宫墙壁， @(64) 为迷宫入口， T(84) 为迷宫出口.
直接运行 FEXsbd3Z.exe 求解，结束后发现闪退，读不出 flag, 改在命令行窗口运行，得到 flag 为 flag{fUn_m4z3_1s_rE4lIy_Re1ax1nG_e8341823} , 提交，解毕.

# 4. 攻防世界 Reversing-x64Elf-100

IDA+F5 得不到 main()函数 ，利用 SHIFT+F12 查找字符串，找到 Enter the password: , 双击进入，对变量名 format 按 X , 找到主函数，再 F5, 得到如下伪代码:

__int64 __fastcall sub_4006FD(__int64 a1)
{
  int i; // [rsp+14h] [rbp-24h]
  __int64 v3[4]; // [rsp+18h] [rbp-20h]

  v3[0] = (__int64)"Dufhbmf";
  v3[1] = (__int64)"pG`imos";
  v3[2] = (__int64)"ewUglpt";
  for ( i = 0; i <= 11; ++i )
  {
    if ( *(char *)(v3[i % 3] + 2 * (i / 3)) - *(char *)(i + a1) != 1 )
      return 1LL;
  }
  return 0LL;
}

__int64 __fastcall main(int a1, char **a2, char **a3)
{
  char s[264]; // [rsp+0h] [rbp-110h] BYREF
  unsigned __int64 v5; // [rsp+108h] [rbp-8h]

  v5 = __readfsqword(0x28u);
  printf("Enter the password: ");
  if ( !fgets(s, 255, stdin) )
    return 0LL;
  if ( (unsigned int)sub_4006FD((__int64)s) )
  {
    puts("Incorrect password!");
    return 1LL;
  }
  else
  {
    puts("Nice!");
    return 0LL;
  }
}

使 sub_4006FD()函数 返回 0, 则得到正确的 flag, 即 if语句 对所有 i 都成立，依托这一点写如下 C 程序:
