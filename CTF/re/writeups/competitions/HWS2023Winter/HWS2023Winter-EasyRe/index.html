小魔改 tea ， key 是随机数，直接通过动调得到

然后解

#include<stdio.h>
#define DELTA 0x61c88647

void tea_encrypt(unsigned int *v, unsigned int *key)
{
    unsigned int l = v[0], r = v[1], sum = 0;
    for (size_t i = 0; i < 32; i++)
    { //进行32次迭代加密，Tea算法作者的建议迭代次数
        l += (((r << 4) ^ (r >> 5)) + r) ^ (sum + key[sum & 3]);
        //printf("%u\n", sum - DELTA);
        sum -= DELTA;                                        //累加Delta的值
        r += (((l << 4) ^ (l >> 5)) + l) ^ (sum + key[(sum >> 11) & 3]); //利用多次双位移和异或将明文与密钥扩散混乱，并将两个明文互相加密
    }
    v[0] = l;
    v[1] = r;
}

void tea_decrypt(unsigned int *v, unsigned int *key)
{
    unsigned int l = v[0], r = v[1], sum = 0;
    sum = -DELTA * 32; // 32次迭代累加后delta的值
    for (size_t i = 0; i < 32; i++)
    {
        r -= (((l << 4) ^ (l >> 5)) + l) ^ (sum + key[(sum >> 11) & 3]);
        //printf("%u\n", sum + DELTA);
        sum += DELTA;
        l -= (((r << 4) ^ (r >> 5)) + r) ^ (sum + key[sum & 3]);
    }
    v[0] = l;
    v[1] = r;
}

int main() {
    unsigned int cipher[] = {
        0xC0B29B34, 0xEF30AF6A, 0x98CCB238, 0x85B6F195, 0xA2480685,
        0xA63D9B59, 0xF191C71E, 0x6790767B
    };

    for (int i = 0; i < 8; i ++) printf("%u,", cipher[i]);
    printf("\n");
    unsigned int key[4] = {
        0x29, 0x4823, 0x18BE, 0x6784
    };

    for (int i = 0; i < 4; i ++ ) {
        tea_decrypt(cipher + 2 * i, key);
    }

    for (int i = 0; i < 8; i ++) printf("%u,", cipher[i]);

}
