#include <stdio.h>

unsigned int MX(unsigned int z, unsigned int y, unsigned int total, unsigned int* key, unsigned int p, unsigned int e) {
    unsigned int temp1 = (z >> 6 ^ y << 4) + (y >> 2 ^ z << 5);
    unsigned int temp2 = (total ^ y) + (key[(p & 3) ^ e] ^ z);
    return temp1 ^ temp2;
}
